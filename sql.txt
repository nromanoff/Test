
create table tags(
id_tag int,
tag_name varchar(50) NOT NULL,
PRIMARY KEY (id_tag)
);

create table topics (
id_topic int,
topic_name varchar(100) NOT NULL,
id_tag int,
PRIMARY KEY (id_topic),
FOREIGN KEY (id_tag) REFERENCES tags(id_tag)
);

select topic_name, tag_name
from topics as [topics] left join tags as [tags] on topics.id_tag = tags.id_tag
group by topic_name, tag_name
